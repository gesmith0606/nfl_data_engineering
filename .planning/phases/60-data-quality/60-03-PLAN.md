    ```yaml
    jobs:
      # ------------------------------------------------------------------
      # Quality Gate: sanity check must pass before any deploy (per D-07)
      # ------------------------------------------------------------------
      quality-gate:
        name: Data Quality Gate
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4

          - name: Set up Python 3.11
            uses: actions/setup-python@v5
            with:
              python-version: "3.11"
              cache: pip
              cache-dependency-path: requirements.txt

          - name: Install Python dependencies
            run: pip install -r requirements.txt

          - name: Run sanity check
            run: |
              python scripts/sanity_check_projections.py \
                --scoring half_ppr \
                --season 2026
            # Exit code 1 on CRITICAL issues -> blocks deployment
    ```

    ```yaml
    deploy-frontend:

      needs: quality-gate
      if: contains(github.event.head_commit.message, 'web/frontend') || github.event_name == 'workflow_dispatch'
    ```

    ```yaml
    deploy-backend:
      runs-on: ubuntu-latest
      needs: quality-gate
    ```
