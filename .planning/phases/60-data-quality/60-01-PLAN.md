    ```python
    def build_roster_mapping(players: dict) -> Dict[str, Dict[str, str]]:
        """Build name -> {team, position} mapping from Sleeper. Per D-03, D-04."""
        mapping: Dict[str, Dict[str, str]] = {}
        skipped_no_team = 0
        for player_id, info in players.items():
            if not isinstance(info, dict):
                continue
            pos = info.get('position')
            if pos not in FANTASY_POSITIONS:
                continue
            team = info.get('team')
            if not team:
                skipped_no_team += 1
                continue
            full_name = info.get('full_name') or ''
            if not full_name:
                first = info.get('first_name', '')
                last = info.get('last_name', '')
                full_name = f"{first} {last}".strip()
            if not full_name:
                continue
            team = SLEEPER_TO_NFLVERSE_TEAM.get(team, team)
            name_key = full_name.lower().strip()
            # Prefer Active players when name collision exists (Pitfall 1 from RESEARCH.md)
            if name_key in mapping and info.get('status') != 'Active':
                continue
            mapping[name_key] = {'team': team, 'position': pos}
        logger.info("Built roster mapping: %d players with teams, %d without", len(mapping), skipped_no_team)
        return mapping
    ```

    **2. Add `update_rosters()` function** (replaces update_teams for combined update per D-03):
    ```python
    def update_rosters(df: pd.DataFrame, roster_mapping: Dict[str, Dict[str, str]]) -> Tuple[pd.DataFrame, pd.DataFrame]:
        """Update recent_team AND position in Gold projections. Per D-03, D-05."""
        df = df.copy()
        changes = []
        for idx, row in df.iterrows():
            name_key = str(row['player_name']).lower().strip()
            mapping = roster_mapping.get(name_key)
            if mapping is None:
                continue
            old_team = row['recent_team']
            new_team = mapping['team']
            old_pos = row['position']
            new_pos = mapping['position']
            if old_team != new_team or old_pos != new_pos:
                change = {
                    'player_name': row['player_name'],
                    'position': new_pos,
                    'old_team': old_team,
                    'new_team': new_team,
                }
                if old_pos != new_pos:
                    change['old_position'] = old_pos
                    change['new_position'] = new_pos
                changes.append(change)
                df.at[idx, 'recent_team'] = new_team
                df.at[idx, 'position'] = new_pos  # D-05: Gold only
        return df, pd.DataFrame(changes)
    ```

    **3. Add `log_changes()` function** (per D-02):
    ```python
    def log_changes(changes_df: pd.DataFrame, log_path: str = 'roster_changes.log') -> None:
        """Append roster changes to persistent log file. Per D-02."""
        timestamp = datetime.now().strftime('%Y-%m-%d %H:%M:%S')
        with open(log_path, 'a') as f:
            f.write(f'\n--- Roster Refresh: {timestamp} ---\n')
            if changes_df.empty:
                f.write('No changes detected.\n')
            else:
                for _, row in changes_df.iterrows():
                    f.write(f"  {row['player_name']} ({row['position']}): "
                            f"{row.get('old_team', '?')} -> {row.get('new_team', '?')}")
                    if 'old_position' in row and pd.notna(row.get('old_position')):
                        f.write(f", pos: {row['old_position']} -> {row['new_position']}")
                    f.write('\n')
    ```
