// web/frontend/src/hooks/use-persistent-chat.ts
export interface UsePersistentChatOptions {
  storageKey?: string;       // default 'advisor:conversation:v1'
  maxMessages?: number;      // default 100 — truncate oldest when exceeded
}

export function usePersistentChat(opts?: UsePersistentChatOptions): {
  messages: UIMessage[];
  sendMessage: (m: { text: string }) => void;
  status: 'idle' | 'streaming' | 'submitted' | 'error';
  error: Error | null;
  clear: () => void;   // wipes localStorage + in-memory messages
}

Internals:
  - On mount (client only): read localStorage[storageKey], JSON.parse, pass to useChat as initialMessages.
  - On messages change: JSON.stringify and write back. Debounce writes with a 250 ms trailing throttle.
  - On clear(): call localStorage.removeItem(storageKey) and setMessages([]).
  - Must be SSR-safe: gate localStorage access behind `typeof window !== 'undefined'` (Next.js 16 strict).
</interfaces>
