<!-- Bronze odds schema (from data/bronze/odds/season=2020/) -->
Columns: game_id, season, week, game_type, home_team, away_team,
         opening_spread, closing_spread, opening_total, closing_total,
         home_moneyline, away_moneyline, nflverse_spread_line, nflverse_total_line
All float64 except game_id (str), home_team/away_team (str), game_type (str).

<!-- Silver output schema (per-team-per-week) -->
Output columns: team, opponent, season, week, game_id, game_type, is_home,
  opening_spread, closing_spread, spread_shift,           # directional (D-02)
  opening_total, closing_total, total_shift,               # symmetric (D-01/D-03)
  spread_move_abs, total_move_abs,                         # symmetric
  spread_magnitude, total_magnitude,                       # ordinal int 0-3, symmetric
  crosses_key_spread, crosses_key_total,                   # bool, symmetric (D-09/D-10/D-11)
  is_steam_move                                            # NaN always (D-15/D-16)

<!-- Sign convention -->
Home rows: directional features as-is from Bronze (positive = home favored)
Away rows: negate opening_spread, closing_spread, spread_shift
Symmetric features: identical for both home and away rows
