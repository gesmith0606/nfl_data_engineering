```typescript
const OL_DEPTH_POSITIONS = new Set(['LT', 'LG', 'C', 'RG', 'RT', 'T', 'G']);

function findOLPlayer(
  roster: RosterPlayer[],
  slot: 'LT' | 'LG' | 'C' | 'RG' | 'RT'
): RosterPlayer | null {
  // Prefer exact depth_chart_position match at depth_chart_order 1.
  const exact = roster
    .filter((p) => p.depth_chart_position === slot)
    .sort((a, b) => (a.depth_chart_order ?? 99) - (b.depth_chart_order ?? 99))[0];
  if (exact) return exact;
  // Fallback: generic T or G at the correct side — acceptable while depth_chart ordering is sparse.
  const generic = roster.find(
    (p) =>
      (slot === 'LT' || slot === 'RT'
        ? p.depth_chart_position === 'T'
        : slot === 'LG' || slot === 'RG'
          ? p.depth_chart_position === 'G'
          : p.depth_chart_position === 'C')
  );
  return generic ?? null;
}

/** Build defensive roster from real API data (replaces buildDefensiveRoster + slotHash). */
function buildDefensiveRosterFromApi(
  rosterResponse: TeamRosterResponse | undefined,
  defenseMetrics: TeamDefenseMetricsResponse | undefined
): Map<string, RatedPlayer | null> {
  const result = new Map<string, RatedPlayer | null>();
  if (!rosterResponse) {
    for (const slot of DEFENSE_SLOTS) result.set(slot.slot, null);
    return result;
  }

  // Group players by their depth_chart_position (DE, DT, NT, OLB, ILB, MLB, CB, FS, SS).
  const byDepth = new Map<string, RosterPlayer[]>();
  for (const p of rosterResponse.players) {
    const dp = p.depth_chart_position ?? p.position;
    if (!byDepth.has(dp)) byDepth.set(dp, []);
    byDepth.get(dp)!.push(p);
  }
  // Sort each depth group by defensive snap% desc, then depth_chart_order asc
  for (const [, group] of byDepth) {
    group.sort((a, b) => {
      const snapDiff = (b.snap_pct_defense ?? 0) - (a.snap_pct_defense ?? 0);
      if (snapDiff !== 0) return snapDiff;
      return (a.depth_chart_order ?? 99) - (b.depth_chart_order ?? 99);
    });
  }

  // Map each defensive slot to candidate depth_chart_positions.
  const slotCandidates: Record<string, string[]> = {
    DE1: ['DE'],
    DE2: ['DE'],
    DT1: ['DT', 'NT'],
    DT2: ['DT', 'NT'],
    LB1: ['OLB', 'LB', 'MLB'],
    LB2: ['ILB', 'LB', 'MLB'],
    LB3: ['OLB', 'ILB', 'LB'],
    CB1: ['CB'],
    CB2: ['CB'],
    SS: ['SS', 'S'],
    FS: ['FS', 'S']
  };

  const used = new Set<string>();
  // Derive a team-wide defensive rating proxy: use overall_def_rating as baseline, nudge by position rank.
  const overallRating = defenseMetrics?.overall_def_rating ?? 70;

  for (const defSlot of DEFENSE_SLOTS) {
    const candidates = slotCandidates[defSlot.slot] ?? [defSlot.pos];
    let picked: RosterPlayer | null = null;
    for (const dp of candidates) {
      const group = byDepth.get(dp) ?? [];
      picked = group.find((p) => !used.has(p.player_id)) ?? null;
      if (picked) break;
    }
    if (!picked) {
      result.set(defSlot.slot, null);
      continue;
    }
    used.add(picked.player_id);

    // Rating: start from team overall_def_rating, add small depth-chart bonus for starters (snap_pct_defense >= 0.5 → +3).
    const starterBonus = (picked.snap_pct_defense ?? 0) >= 0.5 ? 3 : 0;
    const depthPenalty = (picked.depth_chart_order ?? 1) > 1 ? -5 : 0;
    const rating = Math.min(99, Math.max(50, overallRating + starterBonus + depthPenalty));

    result.set(defSlot.slot, {
      player_id: picked.player_id,
      player_name: picked.player_name,
      team: picked.team,
      position: picked.depth_chart_position ?? picked.position,
      projected_points: null,
      injury_status: picked.injury_status ?? picked.status ?? null,
      rating,
      position_rank: null
    });
  }
  return result;
}
```
